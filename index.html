Два шанса на победу. Сетка победителей и сетка проигравших с гранд-финалом между чемпионами обеих сеток.

double_elimination

// Генерация турнирной сетки и запуск турнира
export async function POST(request, { params }) {
  const { id } = await params;
  const tournament = await db.tournament.findUnique({
    where: { id },
    include: { teams: { include: { team: true } } }
  });

  // Подготовка посевных команд
  const seedTeams = tournament.teams.map(tt => ({
    id: tt.id, teamId: tt.teamId, seed: tt.seed
  }));

  // Генерация сетки по выбранному формату
  const bracketMatches = generateBracket(
    tournament.format, seedTeams
  );

  // Создание матчей в БД со связями nextMatchId
  for (const match of bracketMatches) {
    await db.match.create({
      data: { tournamentId: id, ...match, status: 'pending' }
    });
  }

  // Инициализация турнирной таблицы
  await db.tournament.update({
    where: { id },
    data: { status: 'active', startDate: new Date() }
  });
}
